import pygame

#initialize pygame and screen dimensions
pygame.init()
SCREEN_WIDTH, SCREEN_HEIGHT = 500, 500

display_surface = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption('adding images and bg images')
bg_image = pygame.transform.scale(pygame.image.load('background.png').convert(), (SCREEN_WIDTH, SCREEN_HEIGHT))

world_image = pygame.transform.scale(pygame.image.load('world.png').convert_alpha(), (200, 200))
world_rect = world_image.get_rect(center=(SCREEN_WIDTH//2, SCREEN_HEIGHT//2 - 30))

text = pygame.font.Font(None, 30).render('Hello World', True, pygame.Color('black'))
text_rect = text.get_rect(center=(SCREEN_WIDTH//2, SCREEN_HEIGHT//2 + 110))

def game_loop():
  clock = pygame.time.Clock() 
  running = True
  while running:
    for event in pygame.event.get():
      if event.type == pygame.QUIT:
        running = False

    display_surface.blit(bg_image, (0, 0))
    display_surface.blit(world_image, world_rect)
    display_surface.blit(text, text_rect)
    pygame.display.flip()
    clock.tick(30)
  pygame.quit()
if __name__ == '__main__':
  game_loop()
